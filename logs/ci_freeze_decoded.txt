name: Contract Freeze Check

on:
  pull_request:
    paths:
      - "core/common/**"
      - ".github/workflows/contract-freeze-check.yml"

jobs:
  verify-freeze:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set up JDK 17
        uses: actions/setup-java@v4
        with:
          java-version: "17"
          distribution: "temurin"

      - name: Grant execute permission for gradlew
        run: chmod +x gradlew

      - name: Run core/common tests (includes FreezeMarkerTest)
        run: ./gradlew :core:common:test --no-daemon
